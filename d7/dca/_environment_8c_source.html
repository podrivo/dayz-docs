332 protected float GetEnvironmentTemperature()
333 {
334 float temperature;
335 temperature = g_Game.GetMission().GetWorldData().GetBaseEnvTemperature();
336 temperature += Math.AbsFloat(temperature * m_Clouds * GameConstants.ENVIRO_CLOUDS_TEMP_EFFECT);
337
338 if (IsWaterContact())
339 {
340 temperature -= Math.AbsFloat(temperature * GameConstants.ENVIRO_WATER_TEMPERATURE_COEF);
341 }
342
343 if (IsInsideBuilding())
344 {
345 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_COEF);
346 }
347 else if (IsInsideVehicle())
348 {
349 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_VEHICLE_COEF);
350 }
351 else if (IsUnderRoof())
352 {
353 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_UNDERROOF_COEF);
354 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
355 }
356 else
357 {
358 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
359 temperature -= Math.AbsFloat(temperature * m_Fog * GameConstants.ENVIRO_FOG_TEMP_EFFECT);
360 temperature -= GetTemperatureHeightCorrection();
361 }
362
363 // incorporate temperature from temperature sources (buffer)
364 if (Math.AbsFloat(m_UTSAverageTemperature) > 0.001)
365 {
366 temperature += m_UTSAverageTemperature;
367 }
368
369 return temperature;

372 // Calculats wet/drying delta based on player's location and weather
373 float GetWetDelta()
374 {
375 float wetDelta = 0;
376 if ( IsWaterContact() )
377 {
379 if (m_WaterLevel >= WATER_LEVEL_HIGH)
380 {
381 wetDelta = 1;
382 }
383 else if (m_WaterLevel >= WATER_LEVEL_MID && m_WaterLevel < WATER_LEVEL_HIGH)
384 {
385 wetDelta = 0.66;
386 }
387 else if (m_WaterLevel >= WATER_LEVEL_LOW && m_WaterLevel < WATER_LEVEL_MID)
388 {
389 wetDelta = 0.66;
390 }
391 else if (m_WaterLevel >= WATER_LEVEL_NONE && m_WaterLevel < WATER_LEVEL_LOW)
392 {
393 wetDelta = 0.33;
394 }
395 }
396 else if (IsRaining() && !IsInsideBuilding() && !IsUnderRoof() && !IsInsideVehicle())
397 {
399 wetDelta = GameConstants.ENVIRO_WET_INCREMENT * GameConstants.ENVIRO_TICKS_TO_WETNESS_CALCULATION * (m_Rain) * (1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind));
400 }
401 else
402 {
404 float tempEffect = Math.Max(m_PlayerHeat + GetEnvironmentTemperature(), 1.0);
405 if (m_HeatSourceTemp != 0)
406 {
407 tempEffect *= GameConstants.ENVIRO_FIRE_INCREMENT;
408 }
409
410 float weatherEffect = ((1 - (m_Fog * GameConstants.ENVIRO_FOG_DRY_EFFECT))) * (1 - (m_Clouds * GameConstants.ENVIRO_CLOUD_DRY_EFFECT));
411 if (weatherEffect <= 0)
412 {
413 weatherEffect = 1.0;
414 }
415
416 wetDelta = -(GameConstants.ENVIRO_DRY_INCREMENT * weatherEffect * tempEffect);
417 if (!IsInsideBuilding())
418 {
419 wetDelta *= 1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind);
420 }
421 }
422
423 return wetDelta;
424 }

708 protected void ProcessItemsHeat()
709 {
710 float hcHead, hcBody, hcFeet;
711 float hHead, hBody, hFeet;
712
713 float heatComfortAvg;
714 float heatAvg;
715
716 BodyPartHeatProperties(m_HeadParts, GameConstants.ENVIRO_HEATCOMFORT_HEADPARTS_WEIGHT, hcHead, hHead);
717 BodyPartHeatProperties(m_BodyParts, GameConstants.ENVIRO_HEATCOMFORT_BODYPARTS_WEIGHT, hcBody, hBody);
718 BodyPartHeatProperties(m_FeetParts, GameConstants.ENVIRO_HEATCOMFORT_FEETPARTS_WEIGHT, hcFeet, hFeet);
719
720 heatComfortAvg = (hcHead + hcBody + hcFeet) / 3;
721 heatAvg = (hHead + hBody + hFeet) / 3;
722 heatAvg = heatAvg * GameConstants.ENVIRO_ITEM_HEAT_TRANSFER_COEF;
723
724 // heat buffer
725 float applicableHB = 0.0;
726 if (m_UTSAverageTemperature < 0.001)
727 {
728 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
729 if (applicableHB > 0.0)
730 {
731 if (m_HeatBufferTimer > 1.0)
732 {
733 m_Player.GetStatHeatBuffer().Add(Math.Min(EnvTempToCoef(m_EnvironmentTemperature), -0.1) * GameConstants.ENVIRO_PLAYER_HEATBUFFER_DECREASE);
734 }
735 else
736 {
737 m_HeatBufferTimer += GameConstants.ENVIRO_PLAYER_HEATBUFFER_TICK;
738 }
739 }
740 else
741 {
742 m_HeatBufferTimer = 0.0;
743 }
744 }
745 else
746 {
747 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
748 if (m_HeatComfort > PlayerConstants.THRESHOLD_HEAT_COMFORT_MINUS_WARNING)
749 {
750 m_Player.GetStatHeatBuffer().Add(GameConstants.ENVIRO_PLAYER_HEATBUFFER_INCREASE);
751 m_HeatBufferTimer = 0.0;
752 }
753 else
754 {
755 m_HeatBufferTimer = 0.0;
756 }
757 }
758
759 m_HeatComfort = (heatComfortAvg + heatAvg + (GetPlayerHeat() / 100)) + EnvTempToCoef(m_EnvironmentTemperature);
760 if ((m_HeatComfort + applicableHB) < (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
761 {
762 m_HeatComfort += applicableHB;
763 }
764 else
765 {
766 if (m_HeatComfort <= (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
767 {
768 m_HeatComfort = PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01;
769 }
770 }
771
772 m_HeatComfort = Math.Clamp(m_HeatComfort, m_Player.GetStatHeatComfort().GetMin(), m_Player.GetStatHeatComfort().GetMax());
773
774 m_Player.GetStatHeatComfort().Set(m_HeatComfort);
775 }

797 {
798 int attCount;
799
800 EntityAI attachment;
801 ItemBase item;
802
803 pHeatComfort = -1;
804 attCount = m_Player.GetInventory().AttachmentCount();
805
806 for (int attIdx = 0; attIdx < attCount; attIdx++)
807 {
808 attachment = m_Player.GetInventory().GetAttachmentFromIndex(attIdx);
809 if (attachment.IsClothing())
810 {
811 item = ItemBase.Cast(attachment);
812 int attachmentSlot = attachment.GetInventory().GetSlotId(0);
813
815 for (int i = 0; i < pBodyPartIds.Count(); i++)
816 {
817 if (attachmentSlot == pBodyPartIds.Get(i))
818 {
819 float heatIsoMult = 1.0;
820 if (attachmentSlot == InventorySlots.VEST)
821 {
822 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_VEST_WEIGHT;
823 }
824
825 if (attachmentSlot == InventorySlots.BACK)
826 {
827 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_BACK_WEIGHT;
828 }
829
830 pHeatComfort += heatIsoMult * MiscGameplayFunctions.GetCurrentItemHeatIsolation(item);
831
832 // go through any attachments and cargo (only current level, ignore nested containers - they isolate)
833 int inAttCount = item.GetInventory().AttachmentCount();
834 if (inAttCount > 0)
835 {
836 for (int inAttIdx = 0; inAttIdx < inAttCount; inAttIdx++)
837 {
838 EntityAI inAttachment = item.GetInventory().GetAttachmentFromIndex(inAttIdx);
839 ItemBase itemAtt = ItemBase.Cast(inAttachment);
840 if (itemAtt)
841 {
842 pHeat += itemAtt.GetTemperature();
843 }
844 }
845 }
846 if (item.GetInventory().GetCargo())
847 {
848 int inItemCount = item.GetInventory().GetCargo().GetItemCount();
849
850 for (int j = 0; j < inItemCount; j++)
851 {
852 ItemBase inItem;
853 if (Class.CastTo(inItem, item.GetInventory().GetCargo().GetItem(j)))
854 {
855 pHeat += inItem.GetTemperature();
856 }
857 }
858 }
859 }
860 }
861 }
862 }
863
864 pHeatComfort = (pHeatComfort / pBodyPartIds.Count()) * pCoef;
865 pHeat = (pHeat / pBodyPartIds.Count()) * pCoef;
866 }

898 protected void ProcessTemperatureSources()
899 {
900 if (m_UTemperatureSources.Count() == 0)
901 {
902 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add(0);
903
904 return;
905 }
906
907 array<float> utsTemperatures = new array<float>();
908
909 // get temperature from the source (based on distance), save it for min/max filtering
910 for (int i = 0; i < m_UTemperatureSources.Count(); i++)
911 {
912 utsTemperatures.Insert(CalcTemperatureFromTemperatureSource(m_UTemperatureSources[i]));
913 }
914
915 float min = MiscGameplayFunctions.GetMinValue(utsTemperatures);
916 float max = MiscGameplayFunctions.GetMaxValue(utsTemperatures);
917
918 if (max > 0 && min < 0)
919 {
921 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add((max + min) * 0.5);
922 }
923 else
924 {
925 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add(max);
926 }

934 float CalcTemperatureFromTemperatureSource(notnull UTemperatureSource uts)
935 {
936 float distance = vector.Distance(m_Player.GetPosition(), uts.GetPosition());
937 distance = Math.Max(distance, 0.1); //min distance cannot be 0 (division by zero)
938 float temperature = 0;
939
940 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::distance: %1", distance), "Environment");
941
943 if (distance > uts.GetFullRange())
944 {
945 float distFactor = 1 - (distance / uts.GetMaxRange());
946 distFactor = Math.Max(distFactor, 0.0);
947 temperature = uts.GetTemperature() * distFactor;
948 //temperature = Math.Clamp(temperature, uts.GetTemperatureMin(), uts.GetTemperatureMax());
949 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::distFactor: %1", distFactor), "Environment");
950 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperatureXX: %1", temperature), "Environment");
951 }
952 else
953 {
954 temperature = uts.GetTemperature();
955 }
956
957 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperature: %1", temperature), "Environment");
958
959 return temperature;

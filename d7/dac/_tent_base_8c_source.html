739 if (m_State == PACKED)

741 return;
742 }
743

750

752 toggle_off.ToLower();

754 toggle_on.ToLower();
755
756 if (toggle_off == selection || toggle_on == selection)
757 {
758 is_closed = m_OpeningMask & toggle.GetOpeningBit();
759 if (is_closed)

761 SetAnimationPhase(toggle.GetToggleOff(), 0);
762 AddProxyPhysics(toggle.GetToggleOff());
763 SetAnimationPhase(toggle.GetToggleOn(), 1);
764 RemoveProxyPhysics(toggle.GetToggleOn());
765 m_ToggleAnimations.Set(toggle, false);
766 m_IsToggle = true;
767 m_OpeningMask &= ~toggle.GetOpeningBit();
768
769 if (selection.Contains("window"))
770 {
771 ManipulateWindow();
772 }
773
774 if (selection.Contains("entrance") || selection.Contains("door"))
775 {
776 ManipulateEntrance();
777 }
778
779 AnimateCamonetToggle(toggle);
780 }
781 else
782 {
783 SetAnimationPhase(toggle.GetToggleOff(), 1);
784 RemoveProxyPhysics(toggle.GetToggleOff());
785 SetAnimationPhase(toggle.GetToggleOn(), 0);
786 AddProxyPhysics(toggle.GetToggleOn());
787 m_ToggleAnimations.Set(toggle, true);
788 m_IsToggle = false;
789 m_OpeningMask |= toggle.GetOpeningBit();
790
791 if (selection.Contains("window"))
792 {
793 ManipulateWindow();
794 }
795
796 if (selection.Contains("entrance") || selection.Contains("door"))
797 {

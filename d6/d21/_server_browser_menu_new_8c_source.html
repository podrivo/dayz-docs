506 void LoadFavoriteServers()
507 {
508 m_Favorites = new TStringArray;
509
510#ifdef PLATFORM_WINDOWS
511 OnlineServices.GetFavoriteServers(m_Favorites);
512#else
513 GetGame().GetProfileStringList("SB_Favorites", m_Favorites);
514
515 // ignore any ids that do not follow correct IP:PORT format
516 for (int i = 0; i < m_Favorites.Count(); ++i)
517 {
518 string id = m_Favorites[i];
519 array<string> output = new array<string>;
520
521 id.Split(":", output);
522 if (output.Count() != 2)
523 {
524 m_Favorites.Remove(i);
525 --i;
526 }
527 }
528
529 // only handle MAX_FAVORITES on consoles
530 if (m_Favorites.Count() > MAX_FAVORITES)
531 {
532 // favorites are ordered by when they were favorited
533 // so keep most recent favorites and ignore oldest favorites
534 m_Favorites.Invert();
535 m_Favorites.Resize(MAX_FAVORITES);
536 m_Favorites.Invert();
537 }
538#endif
539 }

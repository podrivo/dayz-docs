109 void ScrollToEntry( ServerBrowserEntry entry )
110 {
111 if ( entry )
112 {
113 float x, y;
114 float x_s, y_s;
115 float x_l, y_l;
116
117 Widget root = entry.GetRoot();
118 Widget first_child = root.GetParent().GetChildren();
119 Widget last_child = first_child;
120 while ( last_child )
121 {
122 if ( last_child.GetSibling() )
123 last_child = last_child.GetSibling();
124 else
125 break;
126 }
127
128 root.GetParent().Update();
129 root.Update();
130
131 m_ServerListScroller.GetScreenPos( x, y );
132 m_ServerListScroller.GetScreenSize( x_s, y_s );
133
134 float bottom_pos = y + y_s;
135
136 root.GetScreenPos( x_l, y_l );
137 root.GetScreenSize( x_s, y_s );
138
139 if ( root == first_child )
140 {
141 m_ServerListScroller.VScrollToPos01( 0 );
142 }
143 else if ( root == last_child )
144 {
145 m_ServerListScroller.VScrollToPos01( 1 );
146 }
147 else if ( y_l + y_s >= bottom_pos )
148 {
149 m_ServerListScroller.VScrollToPos( m_ServerListScroller.GetVScrollPos() + y_s );
150 }
151 else if ( y_l <= y )
152 {
153 m_ServerListScroller.VScrollToPos( m_ServerListScroller.GetVScrollPos() - y_s );
154 }
155 }

481 {

483 {
484 return false;
485 }
486

488 {

490 {
491 return false;
492 }
493 }
494
495 if ( m_Filters.m_PreviouslyPlayedFilter.IsSet() )
496 {
497 bool is_visited = g_Game.IsVisited( result.m_HostIp, result.m_HostPort );
498
499 if ( !is_visited && m_Filters.m_PreviouslyPlayedFilter.IsEnabled() )
500 {
501 return false;
502 }
503
504 if ( is_visited && m_Filters.m_PreviouslyPlayedFilter.IsEnabled() )
505 {
506 return false;
507 }
508 }
509
510 return true;

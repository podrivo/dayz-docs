74 static HandEventBase HandEventFactory (HandEventID id, Man p = null, InventoryLocation src = null)

76 switch (id)

81 case HandEventID.DROP: return new HandEventDrop(p, src);
82 case HandEventID.THROW: return new HandEventThrow(p, src);
83 case HandEventID.SWAP: return new HandEventSwap(p, src);

88 case HandEventID.REPLACE: return new HandEventDestroyAndReplaceWithNew(p, src);
89 case HandEventID.REPLACE2: return new HandEventDestroyAndReplaceWithNewElsewhere(p, src);

91 case HandEventID.REPLACED: return new HandEventReplaced(p, src);
92 case HandEventID.ANIMEVENT_CHANGE_HIDE: return HandAnimEventChanged(p, src);
93 case HandEventID.HUMANCOMMAND_ACTION_FINISHED : return HandEventHumanCommandActionFinished(p, src);
94 case HandEventID.HUMANCOMMAND_ACTION_ABORTED : return HandEventHumanCommandActionAborted(p, src);
95 }
96 Error("[hndfsm] HandEventFactory - unregistered hand event with id=" + id);

100 static HandEventBase CreateHandEventFromContext (ParamsReadContext ctx)
101 {
102 int eventID = -1;
103 if (!ctx.Read(eventID))
104 {
105 Error("[hndfsm] CreateHandEventFromContext - cannot read eventID");

107 }
108 Man player;
109 if (!ctx.Read(player))
110 {
111 Error("[hndfsm] CreateHandEventFromContext - cannot read player");
112 return null;
113 }
114 InventoryLocation src;
115 OptionalLocationReadFromContext(src, ctx);
116
117 int animID = -1;
118 if (!ctx.Read(animID))
119 {
120 Error("[hndfsm] CreateHandEventFromContext - cannot read animID");
121 return null;
122 }
123 HandEventBase b = HandEventFactory(eventID, player, src);
124 if (b)
125 {
126 b.m_AnimationID = animID;
127 b.ReadFromContext(ctx);
128 }
129 return b;

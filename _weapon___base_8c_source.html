742 bool FillInnerMagazine( string ammoType = "", int flags = WeaponWithAmmoFlags.CHAMBER )
743 {
744 // Don't try to fill it when there are none
745 if (!HasInternalMagazine(-1))
746 return false;
747
748 // Make sure the given ammoType is actually useable
749 if (ammoType != "")
750 {
751 if (!AmmoTypesAPI.MagazineTypeToAmmoType(ammoType, ammoType))
752 return false;
753 }
754
755
756 bool didSomething = false;
757 int muzzCount = GetMuzzleCount();
758
759 bool ammoRng = ammoType == "";
760 bool ammoFullRng = ammoRng && (flags & WeaponWithAmmoFlags.AMMO_MAG_RNG);
761
762 // No full RNG flag, so pick one random and use only this one
763 if (ammoRng && !ammoFullRng)
764 ammoType = GetRandomChamberableAmmoTypeName(0);
765
766 // Fill the internal magazine
767 for (int i = 0; i < muzzCount; ++i)
768 {
769 int ammoCount = GetInternalMagazineMaxCartridgeCount(i);
770
771 // Decide random quantity when enabled
772 if ( flags & WeaponWithAmmoFlags.QUANTITY_RNG )
773 ammoCount = Math.RandomIntInclusive(0, ammoCount);
774
775 // Only do the things when there is actually ammo to fill
776 if (ammoCount > 0)
777 {
778 // Push in the cartridges
779 for (int j = 0; j < ammoCount; ++j)
780 {
781 // Full random, decide a new one for every cartridge
782 if ( ammoFullRng )
783 ammoType = GetRandomChamberableAmmoTypeName(i);
784
785 PushCartridgeToInternalMagazine(i, 0, ammoType);
786 didSomething = true;
787 }
788 }
789 }
790
791 // Call the chamber method if asked for
792 bool chamber = (flags & WeaponWithAmmoFlags.CHAMBER) || (flags & WeaponWithAmmoFlags.CHAMBER_RNG);
793 if (chamber && FillChamber(ammoType, flags))
794 {
795 didSomething = true;
796 }
797
798 // Does not need any FSM fixing, FSM does not care about inner magazines
799
800 return didSomething;
801 }

810 bool FillChamber( string ammoType = "", int flags = WeaponWithAmmoFlags.CHAMBER )
811 {
812 // Quickly check if there are any chambers we can fill
813 int muzzCount = GetMuzzleCount();
814 bool anyEmpty = false;
815
816 for (int m = 0; m < muzzCount; ++m)
817 {
818 if (IsChamberEmpty(m))
819 {
820 anyEmpty = true;
821 break;
822 }
823 }
824
825 if (!anyEmpty)
826 return false;
827
828 // Make sure the given ammoType is actually useable
829 if (ammoType != "")
830 if (!AmmoTypesAPI.MagazineTypeToAmmoType(ammoType, ammoType))
831 return false;
832
833 // Just so we don't '&' wastefully in a loop
834 bool didSomething = false;
835 bool chamberFullRng = (flags & WeaponWithAmmoFlags.CHAMBER_RNG_SPORADIC);
836 bool chamberRng = (flags & WeaponWithAmmoFlags.CHAMBER_RNG);
837 bool chamber = (flags & WeaponWithAmmoFlags.CHAMBER);
838

840 {
841 int amountToChamber = muzzCount;
842
843 // No need to do this for full rng, as that will roll for every muzzle
844 if (chamberRng)
845 amountToChamber = Math.RandomIntInclusive(0, muzzCount);

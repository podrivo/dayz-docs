384
385 m_animSoundEventsAttack = new array<ref AnimSoundEvent>;
386
387 int soundCount = GetGame().ConfigGetChildrenCount(cfgPath);
388 for(int i = 0; i < soundCount; i++)
389 {
390 string soundName;
391 GetGame().ConfigGetChildName(cfgPath, i, soundName);
392 string soundPath = cfgPath + soundName + " ";
393 AnimSoundEvent soundEvent = new AnimSoundEvent(soundPath);
394 if(soundEvent.IsValid())
395 m_animSoundEventsAttack.Insert(soundEvent);

399 void LoadVegetationSounds()
400 {
401 m_pVegetationSounds = new array<ref VegetationSound>;
402
403 string vegSoundsCfgPath = "CfgVehicles SurvivorBase VegetationSounds ";
404 int vegSoundsCount = GetGame().ConfigGetChildrenCount(vegSoundsCfgPath);
405
406 for (int v = 0; v < vegSoundsCount; ++v)
407 {
408 string vegSoundParamName;
409 GetGame().ConfigGetChildName(vegSoundsCfgPath, v, vegSoundParamName);
410
411 string soundSet = "";
412 GetGame().ConfigGetText(vegSoundsCfgPath + vegSoundParamName + " soundSet", soundSet);
413
414 TIntArray animEventIds = new TIntArray;
415 GetGame().ConfigGetIntArray(vegSoundsCfgPath + vegSoundParamName + " animEventIds", animEventIds);
416
417 SoundParams soundParams = new SoundParams(soundSet);
418
419 if (soundParams.IsValid() && animEventIds.Count() > 0)
420 {
421 m_pVegetationSounds.Insert(new VegetationSound(new SoundObjectBuilder(soundParams), animEventIds));
422 }
423 }
424 }

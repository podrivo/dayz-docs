625 float ComputeHitDirectionAngleDeg(EntityAI source)
626 {
627 vector targetDirection = GetDirection();
628 vector toSourceDirection = (source.GetPosition() - GetPosition());
629
630 targetDirection[1] = 0;
631 toSourceDirection[1] = 0;
632
633 targetDirection.Normalize();
634 toSourceDirection.Normalize();
635
636 float cosFi = vector.Dot(targetDirection, toSourceDirection);
637 vector cross = targetDirection * toSourceDirection;
638
639 float dirAngleDeg = Math.Acos(cosFi) * Math.RAD2DEG;
640 if ( cross[1] < 0 )
641 dirAngleDeg = -dirAngleDeg;
642
643 return dirAngleDeg;
644 }

646 int TranslateHitAngleDegToDirectionIndex(float angleDeg)
647 {
648 if (angleDeg >= -20 && angleDeg <= 20) // front
649 {
650 return 1;
651 }
652 else if (angleDeg < 0) // left
653 {
654 return 2;
655 }
656
657 return 3; // right
658 }

660 int FindComponentDirectionOffset(string component)
661 {
662 const int directionCount = 4;
663
664 int offset = 0;
665 if (component.Length() == 0)
666 {
667 offset = 0;
668 }
669 else if (component == "Zone_Head")
670 {
671 offset = directionCount;
672 }
673 else if (component == "Zone_Chest" || component == "Zone_Legs_Front" || component == "Zone_Spine_Front" || component == "Zone_Neck")
674 {
675 offset = 2 * directionCount;
676 }
677 else
678 {
679 offset = 3 * directionCount;
680 }
681
682 return offset;

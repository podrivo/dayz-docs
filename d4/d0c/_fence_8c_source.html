619 override bool IsPlayerInside( PlayerBase player, string selection )
620 {
621 vector player_pos = player.GetPosition();
622 vector fence_pos = GetPosition();
623 vector ref_dir = GetDirection();
624 ref_dir[1] = 0;
625 ref_dir.Normalize();
626
627 vector x[2];
628 vector b1,b2;
629 GetCollisionBox(x);
630 b1 = x[0];
631 b2 = x[1];
632
633 vector dir_to_fence = fence_pos - player_pos;

637 dir_to_fence.Normalize();

639 vector ref_dir_angle = ref_dir.VectorToAngles();

642
643 vector test_position = test_angles.AnglesToVector() * len;
644
645 if(test_position[0] < b1[0] || test_position[0] > b2[0] || test_position[2] < 0.2 || test_position[2] > 2.2 )

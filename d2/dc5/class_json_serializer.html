Detailed Description

Class for sending RPC over network.

// example sending
void Send()
{
ScriptRPC rpc = new ScriptRPC();

rpc.Write("hello");
array<float> farray = {1.2, 5.6, 8.1};
rpc.Write(farray);
rpc.Send(m_Player, ERPCs.RPC_TEST, true, m_Player.GetIdentity());
}

{

array<float> farray;

ctx.Read(farray);
}

122 override void OnUpdate(float pDt, out DayZPlayerCameraResult pOutResult)
123 {
125 float min;
126 DayZPlayerImplement player = DayZPlayerImplement.Cast(GetGame().GetPlayer());
127 if (player && player.IsPlayerInStance(DayZPlayerConstants.STANCEMASK_RAISEDPRONE))
128 min = CONST_UD_MIN_BACK;
129 else
130 min = CONST_UD_MIN;
131
132 float udAngle = UpdateUDAngle(m_fUpDownAngle, m_fUpDownAngleAdd, min, CONST_UD_MAX, pDt);
133 m_CurrentCameraPitch = udAngle;
134 m_fLeftRightAngle = UpdateLRAngle(m_fLeftRightAngle, CONST_LR_MIN, CONST_LR_MAX, pDt);
135
136 // get model space transform matrix of the gun's eye vector
137 vector matTM[4];
138 HumanItemAccessor hia = m_pPlayer.GetItemAccessor();
139 if(GetCurrentSightEntity())
140 hia.WeaponGetCameraPointBoneRelative(GetCurrentSightEntity(), m_OpticsCamPos, m_OpticsCamDir, m_iBoneIndex, matTM);
141
142 // aim change based on character movement
143 vector aimChangeYPR;
144 float aimChangeX = m_pInput.GetAimChange()[0] * Math.RAD2DEG;
145 float aimChangeY = m_pInput.GetAimChange()[1] * Math.RAD2DEG;
146
147 HumanCommandMove hcm = m_pPlayer.GetCommand_Move();
148 if( hcm )
149 {
150 float speed = hcm.GetCurrentMovementSpeed();
151
152 if( speed > 0 )
153 m_movementTimeAcc += pDt;
154 else
155 m_movementTimeAcc = 0;
156
157 aimChangeX += m_movementAmplitudeX * speed * Math.Sin(Math.PI2 * m_movementFrequencyX * m_movementTimeAcc);
158 aimChangeY += m_movementAmplitudeY * speed * Math.Sin(Math.PI2 * m_movementFrequencyY * m_movementTimeAcc);
159 }
160
161 aimChangeYPR[0] = Math.SmoothCD(aimChangeYPR[0], -(m_dynamicsStrength * aimChangeY), m_velocityYaw, m_dynamicsSmoothTime, 1000, pDt);
162 aimChangeYPR[1] = Math.SmoothCD(aimChangeYPR[1], -(m_dynamicsStrength * aimChangeX), m_velocityPitch, m_dynamicsSmoothTime, 1000, pDt);
163 aimChangeYPR[2] = 0;
164
165 vector dynamics[4];
166 Math3D.YawPitchRollMatrix(aimChangeYPR, dynamics);
167 dynamics[3] = vector.Zero;
168
169 // aiming model offsets
170 vector aimingMatTM[4];
171 hia.WeaponGetAimingModelDirTm(aimingMatTM);
172
173 // final offset matrix
174 Math3D.MatrixMultiply4(dynamics, aimingMatTM, dynamics);
175 Math3D.MatrixMultiply4(dynamics, matTM, pOutResult.m_CameraTM);
176 AdjustCameraParameters(pDt, pOutResult);
177 UpdateBatteryOptics(GetCurrentSightEntity());
178 UpdateCameraNV(PlayerBase.Cast(m_pPlayer));
179
180 if(m_CameraShake)
181 {
182 float x,y;
183 m_CameraShake.Update(pDt, x, y);
184 DayZPlayerImplement.Cast(m_pPlayer).GetAimingModel().SetCamShakeValues(x, y);
185 }
186 }

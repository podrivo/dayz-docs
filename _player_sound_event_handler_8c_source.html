172 {
173 return false;
174 }
175
176 m_CurrentState.Stop();
177 }
178 }
179 m_CurrentState = PlayerSoundEventBase.Cast(requested_state.ClassName().ToType().Spawn());
180 m_CurrentState.InitEx(m_Player, param);

243 bool ProcessIntoIPAddress(out string potential_ip)
244 {
245 bool was_text_changed = false;
246 int length = potential_ip.Length();
247
248 // Filter out incompatible characters
249 for ( int i = length - 1 ; i >= 0; --i )
250 {
251 string character = potential_ip.Get(i);
252
253 if ( character.ToInt() == 0 && character != "0" && character != "." && character != ":" )
254 {
255 potential_ip = potential_ip.SubstringInverted(potential_ip, i, i + 1);
256 was_text_changed = true;
257 }
258 }
259
260 bool text_was_shortened = LimitTextBoxCharacterCount( potential_ip, 21 );
261
262 return (was_text_changed || text_was_shortened);
263 }

281 string GenerateValidIP(string potential_ip)
282 {
283 string correct_ip_format;
284
285 TStringArray array_ip_and_port = new TStringArray;
286 potential_ip.Split( ":", array_ip_and_port );
287
288 string ip_entries;
289 int port;
290
291 if( array_ip_and_port.Count() > 0 )
292 {
293 ip_entries = array_ip_and_port.Get( 0 );
294 TStringArray array_ip = new TStringArray;
295 ip_entries.Split( ".", array_ip );
296
297 string ip_string;
298 int ip_entries_count = array_ip.Count();
299
300 if (ip_entries_count <= 3 || ip_entries_count > 4)
301 return "";
302
303 for ( int i = 0; i < 4; i++ )
304 {
305 int ip_entry = array_ip.Get(i).ToInt();
306
307 // Check if the entry is a valid number
308 if ( ip_entry > 255 || ip_entry < 0 )
309 ip_entry = 0;
310
311 ip_string = ip_string + ip_entry.ToString();
312
313 if ( i < ip_entries_count - 1 )
314 ip_string = ip_string + ".";
315 ;
316 }
317
318 correct_ip_format = ip_string;
319 }
320 else
321 {
322 ip_entries = "";
323 correct_ip_format = ip_entries;
324 }
325
326 if ( correct_ip_format == "" )
327 return ""; // Input string was not recognized as a valid IP address. Thus return an empty string.
328
329 // The input string was recognized as a valid IP address. Thus continue with port number.
330
331 if( array_ip_and_port.Count() > 1 )
332 {
333 port = array_ip_and_port.Get( 1 ).ToInt();
334
335 if ( port > 65535 | port < 0 ) // max port value is 65535
336 port = 2302;
337
338
339 if ( port >= 0 )
340 correct_ip_format = correct_ip_format + ":" + port.ToString();
341
342 }
343 else
344 {
345 port = 2302;
346 correct_ip_format = correct_ip_format + ":" + port.ToString();
347 }
348
349 return correct_ip_format;
350 }
